        async function generateReview(text) {

            const apiKey = ""; 

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;


            const prompt = `Anda adalah seorang reviewer akademis yang berpengalaman. Tugas Anda adalah membaca teks makalah berikut dan memberikan evaluasi singkat, poin-poin utama, serta potensi perbaikan untuk setiap bagian. Fokus pada kejelasan, kekuatan, dan kelemahan dari setiap bagian. Terakhir, berikan ringkasan rekomendasi perbaikan secara umum.


Teks Makalah:

---

${text}

---


Berikan output dalam format JSON dengan struktur yang telah ditentukan. Untuk setiap kunci, berikan ulasan dalam bentuk paragraf singkat.`;


            const payload = {

                contents: [{ role: "user", parts: [{ text: prompt }] }],

                generationConfig: {

                    responseMimeType: "application/json",

                    responseSchema: {

                        type: "OBJECT",

                        properties: {

                            "abstrak": { "type": "STRING", "description": "Evaluasi bagian abstrak." },

                            "pendahuluan_latar_belakang": { "type": "STRING", "description": "Evaluasi gabungan bagian pendahuluan dan latar belakang." },

                            "related_work": { "type": "STRING", "description": "Evaluasi kajian pustaka atau related work." },

                            "metodologi": { "type": "STRING", "description": "Evaluasi metodologi penelitian." },

                            "hasil": { "type": "STRING", "description": "Evaluasi penyajian hasil dan pembahasan." },

                            "kesimpulan": { "type": "STRING", "description": "Evaluasi bagian kesimpulan." },

                            "rekomendasi_perbaikan": { "type": "STRING", "description": "Ringkasan rekomendasi perbaikan secara umum untuk keseluruhan makalah." }

                        },

                        required: ["abstrak", "pendahuluan_latar_belakang", "related_work", "metodologi", "hasil", "kesimpulan", "rekomendasi_perbaikan"]

                    }

                }

            };

            

            let response;

            let retries = 3;

            let delay = 1000;

            while (retries > 0) {

                try {

                    response = await fetch(apiUrl, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    if (response.ok) break;

                } catch (error) {

                    console.error('Fetch error:', error);

                }

                retries--;

                if(retries > 0) await new Promise(resolve => setTimeout(resolve, delay));

                delay *= 2;

            }


            if (!response || !response.ok) {

                throw new Error(`API request failed with status ${response?.status}`);

            }


            const result = await response.json();

            loader.style.display = 'none';


            if (result.candidates && result.candidates[0]?.content?.parts?.[0]) {

                currentReviewData = JSON.parse(result.candidates[0].content.parts[0].text);

                setupNavigation();

                showSectionContent(Object.keys(sections)[0]); // Show first section by default

            } else {

                console.error("Unexpected API response structure:", result);

                showError("Gagal mem-parsing hasil review dari API.");

            }

        }


        function setupNavigation() {

            navigationContainer.innerHTML = '';

            for (const key in sections) {

                const isRecommendation = key === 'rekomendasi_perbaikan';

                const buttonClasses = isRecommendation 

                    ? 'nav-button w-full sm:w-auto mt-2 bg-green-500 text-white font-semibold py-2 px-4 rounded-lg hover:bg-green-600'

                    : 'nav-button bg-white text-gray-700 font-semibold py-2 px-4 rounded-lg hover:bg-gray-100 border border-gray-300';


                const button = document.createElement('button');

                button.className = buttonClasses;

                button.dataset.section = key;

                button.textContent = sections[key].title;

                

                if (isRecommendation) {

                    navigationContainer.appendChild(document.createElement('div')).className = 'w-full'; // Force line break

                    navigationContainer.appendChild(button);

                } else {

                    navigationContainer.appendChild(button);

                }

            }

        }


        navigationContainer.addEventListener('click', (e) => {

            if (e.target.tagName === 'BUTTON') {

                const sectionKey = e.target.dataset.section;

                showSectionContent(sectionKey);

            }

        });


        function showSectionContent(key) {

            if (!currentReviewData || !sections[key]) return;


            // Update active button state

            document.querySelectorAll('.nav-button').forEach(btn => {

                btn.classList.remove('active');

                if (btn.dataset.section === key) {

                    btn.classList.add('active');

                }

            });


            const sectionInfo = sections[key];

            const content = currentReviewData[key];

            

            reviewContentDisplay.innerHTML = `

                <h3 class="card-title">

                    ${sectionInfo.icon}

                    ${sectionInfo.title}

                </h3>

                <div class="card-content">

                    ${content.replace(/\n/g, '<br>')}

                </div>

            `;

            reviewContentDisplay.classList.remove('hidden');

        }
